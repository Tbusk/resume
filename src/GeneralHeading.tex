\newcommand{\rGeneralHeading}[4]{
    \begin{tabular*}{
        \textwidth % whole doc width
        \vspace{4pt}
    }{
        l@{\extracolsep{\fill}}r % separates cols with spacing
    }
    \hspace{4pt}\textbf{#1} & \small#2 \\
    \vspace{2pt}\hspace{4pt}\textit{\small #3} & \textit{\small #4} \\
    \end{tabular*}\vspace{-6pt}
}
